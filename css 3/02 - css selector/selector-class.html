selector css selector class css

  1. Selector =
    2. [class="header"] memilih semua elemen dengan class yang sama dengan "header" 
                        [class="header"] {
                        color: red;
                    }
  2. Selector *=
    3. [id*="col-"] memilih semua elmen dengan id yang mengandung kata "col-" 
                        [id*="col-"] {
                        color: red;
                    }
  3. Selector ^=
    4. [href^="https://"] memilih semua atribut href yang dimulai dengan kata "https://" 
                        [href^="https://"] {
                        color: red;
                    }
  4. Selector $=
    5. [src$=".png"] memilih semua atribut src yang berakhir dengan kata ".png" 
                        [src$=".png"] {
                        color: red;
                    }
  5. Selector ~=
    6. [rel~="tag"] memilih semua stribut rel yang mengandung kata "tag" 
                        [rel~="tag"] {
                        color: red;
                    }
  6. Selector atribut tertentu []
    7. [required] memilih elemen dengan atribut required 
                        [required] {
                        color: red;
                    }
