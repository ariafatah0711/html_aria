jendela dialog confirm() 
                confirm() = untuk mengonfirmasi tindakan pengguna dengan dua pilihan "ok" dan "batal"
                    - confirm("apakah ada ingin menghapus");
                    - function result() {
                        var yakin = confirm("apakah anda ingin menambahkan paragraph");
                        if (yakin) {
                            var newText = document.createElement("p");
                            newText.textContent =
                              "ini adalah paragraph, dan dapat ditampilkan hanya jika pengguna mengetik ok";
                            document.body.appendChild(newText);
                        } else {
                          // tindakan jika nilai false
                        }
                      }
                      
                      result();

                    var paragraphs = document.querySelectorAll("p");
                    if (paragraphs.length > 0) {
                        paragraphs[paragraphs.length - 1].remove();
                    }
                    }
