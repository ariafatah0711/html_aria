                confirm() = untuk mengonfirmasi tindakan pengguna dengan dua pilihan "ok" dan "batal"
                    - confirm("apakah ada ingin menghapus");
                    - function result() {
                        var yakin = confirm("apakah anda ingin menambahkan paragraph");
                        if (yakin) {
                          document.writeln("<p>ini adalah paragraph, dan dapat ditampilkan hanya jika pengguna mengetik ok</p>");
                        } else {
                          document.write("");
                        }
                      }
