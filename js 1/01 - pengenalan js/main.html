cara penulisan kode JavaScript 
                1. penulisan dengan embed
                <head>
                    <script>isi script</script>
                </head>
                <body>
                    <script>isi script</script>
                </body>

                2. penulisan dengan inline
                <a href="#" onclick="alert('output_alert')">onclick="alert"</a>

                3. penulisan external
                <script src="script.js"></script>
